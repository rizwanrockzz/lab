# BINARY SEARCH

##program : 

def binarySearch(arr,n,key):
    
    low = 0
    high = n-1
    
    while(low<=high):
        
        mid = (low+high)//2
        
        if (arr[mid] == key):
            return mid
            
            
        elif (arr[mid] > key):
            high = mid-1
            
        else:
            low = mid+1

    return -1


n = int(input("No of elements : "))
arr = []

for i in range(n):
    e=int(input())
    arr.append(e)

arr.sort() 
print(arr)    

key = int(input("Enter the key : "))

ans = binarySearch(arr,n,key)

if ans!=-1:
    print("{} is present at location {}".format(key,ans))
    
else:
    print("Element not found")

# PRIMICITY

## Program : 

def prime(num):
    flag = True
    for i in range(2, num):
        if (num % i) == 0:
            flag = False
            break
    return flag

def sub_lists (l):
    lists = [[]]
    for i in range(len(l) + 1):
        for j in range(i):
            lists.append(l[j: i])
    return lists


n = int(input("Enter N : "))
k = int(input("Enter K : "))

arr = []

    arr.append(e)
    
print(arr)
temp = sub_lists(arr)
print(temp)

count=[]

for i in temp:
